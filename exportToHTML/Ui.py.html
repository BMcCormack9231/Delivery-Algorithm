def time_query_all_packages(truck1, truck2, truck3):
    print('Enter any time (format "HH:MM") to check the status of packages on each truck at that timestamp')
    prompt = input('> ')
    try:
        time_format = "%H:%M"
        time_object = datetime.strptime(prompt, time_format)
        time_delta = timedelta(hours=time_object.hour, minutes=time_object.minute)
        if time_delta < timedelta(hours=8):
            print("Error: The operation does not start until 08:00")
            return
        print(f"Displaying all packages at timestamp {prompt}")
        print("Truck 1 packages:")
        for package in truck1.truck_packages:   # O(n)
            print(package.get_status(time_delta))
        print("Truck 2 packages:")
        for package in truck2.truck_packages:   # O(n)
            print(package.get_status(time_delta))
        print("Truck 3 packages:")
        for package in truck3.truck_packages:   # O(n)
            print(package.get_status(time_delta))
    except ValueError:
        print("Error: invalid input")
